var counter = 1; function calculate() { // Get the input values from the form var fz = document.getElementById("fz").value; var fn = document.getElementById("fn").value; var dc = document.getElementById("dc").value; var z = document.getElementById("z").value; var Vc = document.getElementById("Vc").value; // Calculate n var n = Vc * 1000 / (Math.PI * dc); console.log() // Calculate Vf var Vf; if (fn != 0) { Vf = fn * n; } else { Vf = fz * z * n; } // Round up the sum to two decimal places n = n.toFixed(2); Vf = Vf.toFixed(2); // Store the result in an array var results = []; results.push(n); // Display the results var resultElement = document.getElementById("result"); results.forEach(function(result) { var resultItem = document.createElement("div"); resultItem.innerHTML = "Tool " + counter + ": " + "n = " + n + "; Vf = " + Vf; resultElement.appendChild(resultItem); counter++; }); // Return false to prevent the form from being submitted return false; }
